' ==========================================
' 1. REGISTER
' ==========================================

@startuml
!pragma messenger true
title Register Sequence

actor "User"
participant "AuthenticationController"
participant "AuthenticationService"
participant "PasswordEncoder"
participant "ApartmentAccessCodeRepository"
participant "ApartmentAccessCodeService"
participant "UserRepository"
participant "JwtService"

"User" -> "AuthenticationController": POST /api/v1/auth/register
activate "AuthenticationController"
"AuthenticationController" -> "AuthenticationService": register(request)
activate "AuthenticationService"
"AuthenticationService" -> "PasswordEncoder": encode(password)
activate "PasswordEncoder"
"PasswordEncoder" --> "AuthenticationService": encodedPassword
deactivate "PasswordEncoder"

alt request.apartmentId != null
    "AuthenticationService" -> "ApartmentAccessCodeRepository": findByEmailAndApartmentId(email, apartmentId)
    activate "ApartmentAccessCodeRepository"
    "ApartmentAccessCodeRepository" --> "AuthenticationService": Valid AccessCode
    deactivate "ApartmentAccessCodeRepository"
    "AuthenticationService" -> "AuthenticationService": setRole(RESIDENT)
else No Apartment ID
    "AuthenticationService" -> "AuthenticationService": setRole(request.role)
end

"AuthenticationService" -> "UserRepository": save(user)
activate "UserRepository"
"UserRepository" --> "AuthenticationService": savedUser
deactivate "UserRepository"

alt Valid AccessCode exists
    "AuthenticationService" -> "ApartmentAccessCodeService": linkUserToApartment(savedUser, apartment)
    activate "ApartmentAccessCodeService"
    "ApartmentAccessCodeService" --> "AuthenticationService": void
    deactivate "ApartmentAccessCodeService"
    
    "AuthenticationService" -> "AuthenticationService": mark code inactive
    "AuthenticationService" -> "ApartmentAccessCodeRepository": save(validCode)
    activate "ApartmentAccessCodeRepository"
    "ApartmentAccessCodeRepository" --> "AuthenticationService": saved
    deactivate "ApartmentAccessCodeRepository"
end

"AuthenticationService" -> "JwtService": generateToken(user)
activate "JwtService"
"JwtService" --> "AuthenticationService": token
deactivate "JwtService"
"AuthenticationService" --> "AuthenticationController": AuthResponse(token)
deactivate "AuthenticationService"
"AuthenticationController" --> "User": 200 OK
deactivate "AuthenticationController"
@enduml
